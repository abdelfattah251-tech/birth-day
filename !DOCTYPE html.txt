        function checkPassword() {
            const input = document.getElementById('passwordInput').value;

                checkPassword();

        function createHearts() {
            const heartsContainer = document.getElementById('hearts');
            for (let i = 0; i < 15; i++) {
                setTimeout(() => {
                    const heart = document.createElement('div');
                    heart.className = 'heart';
                    heart.textContent = '‚ù§Ô∏è';
                    heart.style.left = Math.random() * 100 + '%';
                    heart.style.animationDuration = (Math.random() * 3 + 4) + 's';
                    heart.style.animationDelay = Math.random() * 2 + 's';
                    heartsContainer.appendChild(heart);
                }, i * 200);
            }
        }

        function playSong() {
            const cat = document.getElementById('singingCat');
            const lyricsDiv = document.getElementById('lyrics');
            
            cat.style.display = 'block';
            
            // Create speech synthesis
            const lyrics = [
                "Happy Birthday to you",
                "Happy Birthday to you",
                "Happy Birthday dear love",
                "Happy Birthday to you",
                "Meow!"
            ];
            
            let index = 0;
            
            function singLine() {
                if (index < lyrics.length) {
                    lyricsDiv.textContent = "üéµ " + lyrics[index] + " üéµ";
                    
                    // Use Web Speech API to sing
                    const utterance = new SpeechSynthesisUtterance(lyrics[index]);
                    utterance.rate = 0.8; // Slower for singing
                    utterance.pitch = 1.2; // Higher pitch
                    utterance.volume = 1;
                    
                    utterance.onend = function() {
                        index++;
                        if (index < lyrics.length) {
                            setTimeout(singLine, 500);
                        } else {
                            setTimeout(() => {
                                lyricsDiv.textContent = "üíï I love you so much! üíï";
                                const loveMsg = new SpeechSynthesisUtterance("I love you so much!");
                                loveMsg.rate = 0.9;
                                loveMsg.pitch = 1.3;
                                speechSynthesis.speak(loveMsg);
                            }, 1000);
                        }
                    };
                    
                    speechSynthesis.speak(utterance);
                }
            }
            
            singLine();
        }
